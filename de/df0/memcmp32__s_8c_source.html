73  const uint32_t *src, rsize_t smax, int *diff)
74 {
75  /*
76  * must be able to return the diff
77  */
78  if (diff == NULL) {
79  invoke_safe_mem_constraint_handler("memcmp32_s: diff is null",
80  NULL, ESNULLP);
81  return (RCNEGATE(ESNULLP));
82  }
83  *diff = -1; /* default diff */
84 
85 
86  if (dest == NULL) {
87  invoke_safe_mem_constraint_handler("memcmp32_s: dest is null",
88  NULL, ESNULLP);
89  return (RCNEGATE(ESNULLP));
90  }
91 
92  if (src == NULL) {
93  invoke_safe_mem_constraint_handler("memcmp32_s: src is null",
94  NULL, ESNULLP);
95  return (RCNEGATE(ESNULLP));
96  }
97 
98  if (dmax == 0) {
99  invoke_safe_mem_constraint_handler("memcmp32_s: dmax is 0",
100  NULL, ESZEROL);
101  return (RCNEGATE(ESZEROL));
102  }
103 
104  if (dmax > RSIZE_MAX_MEM32 || smax > RSIZE_MAX_MEM32) {
105  invoke_safe_mem_constraint_handler("memcmp32_s: dmax/smax exceeds max",
106  NULL, ESLEMAX);
107  return (RCNEGATE(ESLEMAX));
108  }
109 
110  if (smax == 0) {
111  invoke_safe_mem_constraint_handler("memcmp32_s: smax is 0",
112  NULL, ESZEROL);
113  return (RCNEGATE(ESZEROL));
114  }
115 
116  if (smax > dmax) {
117  invoke_safe_mem_constraint_handler("memcmp32_s: smax exceeds dmax",
118  NULL, ESNOSPC);
119  return (RCNEGATE(ESNOSPC));
120  }
121 
122  /*
123  * no need to compare the same memory
124  */
125  if (dest == src) {
126  *diff = 0;
127  return (RCNEGATE(EOK));
128  }
129 
130  /*
131  * now compare src to dest
132  */
133  *diff = 0;
134  while (dmax != 0 && smax != 0) {
135  if (*dest != *src) {
136  *diff = *dest - *src;
137  break;
138  }
139 
140  dmax--;
141  smax--;
142 
143  dest++;
144  src++;
145  }
146 
147  return (RCNEGATE(EOK));
148 }
