74 memcpy16_s (uint16_t *dest, rsize_t dmax, const uint16_t *src, rsize_t smax)
75 {
76  if (dest == NULL) {
77  invoke_safe_mem_constraint_handler("memcpy16_s: dest is NULL",
78  NULL, ESNULLP);
79  return (RCNEGATE(ESNULLP));
80  }
81 
82  if (dmax == 0) {
83  invoke_safe_mem_constraint_handler("memcpy16_s: dmax is 0",
84  NULL, ESZEROL);
85  return (RCNEGATE(ESZEROL));
86  }
87 
88  if (dmax > RSIZE_MAX_MEM16 || smax > RSIZE_MAX_MEM16) {
89  invoke_safe_mem_constraint_handler("memcpy16_s: dmax/smax exceeds max",
90  NULL, ESLEMAX);
91  return (RCNEGATE(ESLEMAX));
92  }
93 
94  if (smax == 0) {
95  mem_prim_set16(dest, dmax, 0);
96  invoke_safe_mem_constraint_handler("memcpy16_s: smax is 0",
97  NULL, ESZEROL);
98  return (RCNEGATE(ESZEROL));
99  }
100 
101  if (smax > dmax) {
102  mem_prim_set16(dest, dmax, 0);
103  invoke_safe_mem_constraint_handler("memcpy16_s: smax exceeds dmax",
104  NULL, ESNOSPC);
105  return (RCNEGATE(ESNOSPC));
106  }
107 
108  if (src == NULL) {
109  mem_prim_set16(dest, dmax, 0);
110  invoke_safe_mem_constraint_handler("memcpy16_s: src is NULL",
111  NULL, ESNULLP);
112  return (RCNEGATE(ESNULLP));
113  }
114 
115  /*
116  * overlap is undefined behavior, do not allow
117  */
118  if( ((dest > src) && (dest < (src+smax))) ||
119  ((src > dest) && (src < (dest+dmax))) ) {
120  mem_prim_set16(dest, dmax, 0);
121  invoke_safe_mem_constraint_handler("memcpy16_s: overlap undefined",
122  NULL, ESOVRLP);
123  return (RCNEGATE(ESOVRLP));
124  }
125 
126  /*
127  * now perform the copy
128  */
129  mem_prim_move16(dest, src, smax);
130 
131  return (RCNEGATE(EOK));
132 }
