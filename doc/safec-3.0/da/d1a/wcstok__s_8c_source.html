160 wcstok_s(wchar_t *restrict dest, rsize_t *restrict dmax,
161  const wchar_t *restrict delim, wchar_t **restrict ptr)
162 {
163  const wchar_t *pt;
164  wchar_t *ptoken;
165  rsize_t dlen;
166  rsize_t slen;
167 
168  if (unlikely(dmax == NULL)) {
169  invoke_safe_str_constraint_handler("wcstok_s: dmax is NULL",
170  NULL, ESNULLP);
171  return (NULL);
172  }
173 
174  if (unlikely(*dmax == 0)) {
175  invoke_safe_str_constraint_handler("wcstok_s: dmax is 0",
176  NULL, ESZEROL);
177  return (NULL);
178  }
179 
180  if (unlikely(*dmax > RSIZE_MAX_WSTR)) {
181  invoke_safe_str_constraint_handler("wcstok_s: dmax exceeds max",
182  NULL, ESLEMAX);
183  return (NULL);
184  }
185 
186  if (unlikely(delim == NULL)) {
187  invoke_safe_str_constraint_handler("wcstok_s: delim is null",
188  NULL, ESNULLP);
189  return (NULL);
190  }
191 
192  if (unlikely(ptr == NULL)) {
193  invoke_safe_str_constraint_handler("wcstok_s: ptr is null",
194  NULL, ESNULLP);
195  return (NULL);
196  }
197 
198  /* if the source was NULL, use the tokenizer context */
199  if (dest == NULL) {
200  dest = *ptr;
201  }
202 
203  /*
204  * scan dest for a delimiter
205  */
206  dlen = *dmax;
207  ptoken = NULL;
208  while (*dest != L'\0' && !ptoken) {
209 
210  if (unlikely(dlen == 0)) {
211  *ptr = NULL;
212  *dmax = 0;
213  *dest = L'\0';
214  invoke_safe_str_constraint_handler(
215  "wcstok_s: dest is unterminated",
216  NULL, ESUNTERM);
217  return (NULL);
218  }
219 
220  /*
221  * must scan the entire delimiter list
222  * ISO should have included a delimiter string limit!!
223  */
224  slen = STRTOK_DELIM_MAX_LEN;
225  pt = delim;
226  while (*pt != L'\0') {
227 
228  if (unlikely(slen == 0)) {
229  *ptr = NULL;
230  *dmax = 0;
231  *dest = L'\0';
232  invoke_safe_str_constraint_handler(
233  "wcstok_s: delim is unterminated",
234  NULL, ESUNTERM);
235  return (NULL);
236  }
237  slen--;
238 
239  if (*dest == *pt) {
240  ptoken = NULL;
241  break;
242  } else {
243  pt++;
244  ptoken = dest;
245  }
246  }
247  dest++;
248  dlen--;
249  }
250 
251  /*
252  * if the beginning of a token was not found, then no
253  * need to continue the scan.
254  */
255  if (ptoken == NULL) {
256  *dmax = dlen;
257  return (ptoken);
258  }
259 
260  /*
261  * Now we need to locate the end of the token
262  */
263  while (*dest != L'\0') {
264 
265  if (unlikely(dlen == 0)) {
266  *ptr = NULL;
267  *dmax = 0;
268  *dest = L'\0';
269  invoke_safe_str_constraint_handler(
270  "wcstok_s: dest is unterminated",
271  NULL, ESUNTERM);
272  return (NULL);
273  }
274 
275  slen = STRTOK_DELIM_MAX_LEN;
276  pt = delim;
277  while (*pt != L'\0') {
278 
279  if (unlikely(slen == 0)) {
280  *ptr = NULL;
281  *dmax = 0;
282  *dest = L'\0';
283  invoke_safe_str_constraint_handler(
284  "wcstok_s: delim is unterminated",
285  NULL, ESUNTERM);
286  return (NULL);
287  }
288  slen--;
289 
290  if (*dest == *pt) {
291  /*
292  * found a delimiter, set to null
293  * and return context ptr to next char
294  */
295  *dest = L'\0';
296  *ptr = (dest + 1); /* return pointer for next scan */
297  *dmax = dlen - 1; /* account for the nulled delimiter */
298  return (ptoken);
299  } else {
300  /*
301  * simply scanning through the delimiter string
302  */
303  pt++;
304  }
305  }
306  dest++;
307  dlen--;
308  }
309 
310  *dmax = dlen;
311  return (ptoken);
312 }
313 EXPORT_SYMBOL(wcstok_s)
