68 EXPORT wint_t _dec_w16(wchar_t *src) {
69  wint_t s1 = src[0];
70  /*if (unlikely(s1 >= 0xd800 && s1 < 0xdc00)) {*/
71  if (unlikely((s1 & 0xfc00) == 0xd800)) {
72  wint_t s2 = src[1];
73  if (likely((s2 & 0xfc00) == 0xdc00)) {
74  return 0x10000 + ((s1 & 0x3ff) << 10) + (s2 & 0x3ff);
75  } else {
76  invoke_safe_str_constraint_handler("wcsnorm_s: "
77  "illegal surrogate pair",
78  NULL, EILSEQ);
79  return 0;
80  }
81  } else {
82  return *src;

87 /* Note that we can generate two versions of the tables. The old format as
88  * used in Unicode::Normalize, and the new 3x smaller NORMALIZE_IND_TBL cperl
89  * variant, as used here and in cperl core since 5.27.2.
90  */
91 static int
92 _decomp_canonical_s(wchar_t *dest, rsize_t dmax, wint_t cp)
93 {
94 #ifndef NORMALIZE_IND_TBL
95  /* the old big format */
96  if (unlikely(dmax < 5)) {
97  return -ESNOSPC;
98  } else {
99  const wchar_t ***plane = UNWIF_canon[cp >> 16];
100  if (! plane) {
101  return 0;
102  } else {
103  const wchar_t **row = plane[(cp >> 8) & 0xff];
104  if (row) {
105  const wchar_t *s = row[cp & 0xff];
106  if (s) {
107  const int c = wcslen(row[cp & 0xff]);
108  wmemcpy_s(dest, dmax, row[cp & 0xff], c);
109  return c;
110  } else {
111  *dest = 0;
112  return 0;
113  }
114  }
115  else {
116  *dest = 0;
117  return 0;
118  }
119  }
120  }
121 #else
122  /* the new format generated with cperl Unicode-Normalize/mkheader -uni -ind */
123  const UNWIF_canon_PLANE_T **plane, *row;
124  if (unlikely(dmax < 5)) {
125  *dest = 0;
126  return -ESNOSPC;
127  }
128  plane = UNWIF_canon[cp >> 16];
129  if (!plane) { /* Only the first 3 of 16 are filled */
130  return 0;
131  }
132  row = plane[(cp >> 8) & 0xff];
133  if (row) { /* the first row is pretty filled, the rest very sparse */
134  const UNWIF_canon_PLANE_T vi = row[cp & 0xff];
135  if (!vi)
136  return 0;
137  else {
138  /* value => length and index */
139  const int l = UNWIF_canon_LEN(vi);
140  const int i = UNWIF_canon_IDX(vi);
141  const wchar_t* tbl = (const wchar_t*)UNWIF_canon_tbl[l-1];
142 #if 0 && defined(DEBUG)
143  printf("U+%04X vi=0x%x (>>12, &fff) => TBL(%d)|%d\n", cp, vi, l, i);
144 #endif
145 #if SIZEOF_WCHAR_T > 2
146  assert(l>0 && l<=4);
147  /* (917,762,227,36) */
148  assert((l==1 && i<917) || (l==2 && i<762) || (l==3 && i<227) ||
149  (l==4 && i<36) || 0);
150  assert(dmax > 4);
151 #endif
152  /* TODO: on sizeof(wchar_t)==2 pre-converted to surrogate pairs */
153  memcpy(dest, &tbl[i*l], l*sizeof(wchar_t)); /* 33% perf */
154  dest[l] = L'\0';
155  return l;
156  }
157  }
158  else {
159  return 0;
160  }
161 #endif
162 }

173 static int
174 _decomp_compat_s(wchar_t *dest, rsize_t dmax, wint_t cp)
175 {
176 #ifndef NORMALIZE_IND_TBL
177  /* the old big format */
178  if (unlikely(dmax < 19)) {
179  *dest = 0;
180  return -ESNOSPC;
181  }
182  else {
183  const wchar_t ***plane = UNWF_compat[cp >> 16];
184  if (! plane) {
185  return 0;
186  } else {
187  const wchar_t **row = plane[(cp >> 8) & 0xff];
188  if (row) {
189  const wchar_t *s = row[cp & 0xff];
190  if (s) {
191  const int c = wcslen(row[cp & 0xff]);
192  wmemcpy_s(dest, dmax, row[cp & 0xff], c);
193  dest[c] = L'\0';
194  return c;
195  } else {
196  *dest = 0;
197  return 0;
198  }
199  }
200  else {
201  *dest = 0;
202  return 0;
203  }
204  }
205  }
206 #else
207  /* the new format generated with cperl Unicode-Normalize/mkheader -uni -ind */
208  const UNWIF_compat_PLANE_T **plane, *row;
209  plane = UNWIF_compat[cp >> 16];
210  if (!plane) { /* Only the first 3 of 16 are filled */
211  return 0;
212  }
213  row = plane[(cp >> 8) & 0xff];
214  if (row) { /* the first row is pretty filled, the rest very sparse */
215  const UNWIF_compat_PLANE_T vi = row[cp & 0xff];
216  if (!vi)
217  return 0;
218  else if (unlikely(vi == (uint16_t)-1)) { /* overlong */
219  UNWIF_compat_exc_t *e;
220  assert(UNWIF_compat_exc_size);
221 #if 1
222  e = (UNWIF_compat_exc_t *)bsearch(&cp, &UNWIF_compat_exc,
223  UNWIF_compat_exc_size, sizeof(UNWIF_compat_exc[0]), _bsearch_exc);
224  if (e) {
225  size_t l = wcslen(e->v);
226  memcpy(dest, e->v, (l+1)*sizeof(wchar_t)); /* incl \0 */
227  return (int)l;
228  }
229 #else
230  for (e=(UNWIF_compat_exc_t*)UNWIF_compat_exc; e->cp; e++) {
231  if (cp == e->cp) {
232  size_t l = wcslen(e->v);
233  if (unlikely(dmax < l)) {
234  *dest = 0;
235  return -(ESNOSPC);
236  }
237  /*assert(l < dmax);*/
238  memcpy(dest, e->v, (l+1)*sizeof(wchar_t)); /* incl \0 */
239  return (int)l;
240  }
241  else if (cp < e->cp)
242  break;
243  }
244 #endif
245  return 0;
246  }
247  else {
248  /* value => length and index */
249  const int l = UNWIF_compat_LEN(vi);
250  const int i = UNWIF_compat_IDX(vi);
251  const wchar_t* tbl = (const wchar_t*)UNWIF_compat_tbl[l-1];
252 #if 0 && defined(DEBUG)
253  printf("U+%04X vi=0x%x (>>12, &&fff) => TBL(%d)|%d\n", cp, vi, l, i);
254 #endif
255  if (unlikely(dmax < (rsize_t)l)) {
256  *dest = 0;
257  return -(ESNOSPC);
258  }
259  /* TODO: on sizeof(wchar_t)==2 pre-converted to surrogate pairs */
260  memcpy(dest, &tbl[i*l], l*sizeof(wchar_t));
261  dest[l] = L'\0';
262  return l;
263  }
264  }
265  else {
266  return 0;
267  }
268 #endif
269 }

272 static int
273 _decomp_hangul_s(wchar_t *dest, rsize_t dmax, wint_t cp)
274 {
275  wint_t sindex = cp - Hangul_SBase;
276  wint_t lindex = sindex / Hangul_NCount;
277  wint_t vindex = (sindex % Hangul_NCount) / Hangul_TCount;
278  wint_t tindex = sindex % Hangul_TCount;
279 
280  if (unlikely(dmax < 4)) {
281  return -(ESNOSPC);
282  }
283 
284  dest[0] = lindex + Hangul_LBase;
285  dest[1] = vindex + Hangul_VBase;
286  if (tindex) {
287  dest[2] = tindex + Hangul_TBase;
288  dest[3] = 0;
289  return 3;
290  } else {
291  dest[2] = 0;
292  return 2;
293  }
294 }

304 EXPORT int
305 _decomp_s(wchar_t *restrict dest, rsize_t dmax, const wint_t cp, const bool iscompat)
306 {
307  /*assert(dmax > 4);*/
308 
309  /* The costly is_HANGUL_cp_high(cp) checks also all composing chars.
310  Hangul_IsS only for the valid start points. Which we can do here. */
311  if (Hangul_IsS(cp)) {
312  return _decomp_hangul_s(dest, dmax, cp);
313  }
314  else {
315 #ifdef HAVE_NORM_COMPAT
316  return iscompat ? _decomp_compat_s(dest, dmax, cp)
317  : _decomp_canonical_s(dest, dmax, cp);
318 #else
319  (void)iscompat;
320  return _decomp_canonical_s(dest, dmax, cp);
321 #endif
322  }
323 }

332 /* rc = wcsnorm_reorder_s(tmp, len+1, dest); */
333 static int _compare_cc(const void *a, const void *b)
334 {
335  int ret_cc;
336  ret_cc = ((UNWIF_cc*) a)->cc - ((UNWIF_cc*) b)->cc;
337  if (ret_cc)
338  return ret_cc;
339 
340  return ( ((UNWIF_cc*) a)->pos > ((UNWIF_cc*) b)->pos )
341  - ( ((UNWIF_cc*) a)->pos < ((UNWIF_cc*) b)->pos );

344 static wint_t _composite_cp(wint_t cp, wint_t cp2)
345 {
346  const UNWIF_complist_s ***plane, **row, *cell;
347 
348  if (unlikely(!cp2)) {
349  invoke_safe_str_constraint_handler("wcsnorm_compose_s: "
350  "cp is 0",
351  NULL, ESZEROL);
352  return 0;
353  }
354  if (unlikely((_UNICODE_MAX < cp) || (_UNICODE_MAX < cp2))) {
355  return -(ESLEMAX);
356  }
357 
358  if (Hangul_IsL(cp) && Hangul_IsV(cp2)) {
359  wint_t lindex = cp - Hangul_LBase;
360  wint_t vindex = cp2 - Hangul_VBase;
361  return(Hangul_SBase + (lindex * Hangul_VCount + vindex) *
362  Hangul_TCount);
363  }
364  if (Hangul_IsLV(cp) && Hangul_IsT(cp2)) {
365  wint_t tindex = cp2 - Hangul_TBase;
366  return (cp + tindex);
367  }
368  plane = UNWIF_compos[cp >> 16];
369  if (!plane) /* only 3 of 16 are defined */
370  return 0;
371  row = plane[(cp >> 8) & 0xff];
372  if (!row) /* the zero plane is pretty filled, the others sparse */
373  return 0;
374  cell = row[cp & 0xff];
375  if (!cell)
376  return 0;
377  /* no indirection here, but search in the composition lists */
378  /* only 16 lists 011099-01d1bc need uint32, the rest can be short, uint16 */
379  /* TODO: above which length is bsearch faster?
380  But then we'd need to store the lengths also */
381  if (likely(cp < UNWIF_COMPLIST_FIRST_LONG)) {
382  UNWIF_complist_s *i;
383  for (i = (UNWIF_complist_s *)cell; i->nextchar; i++) {
384  if ((uint16_t)cp2 == i->nextchar)
385  return (wint_t)(i->composite);
386  else if ((uint16_t)cp2 < i->nextchar) /* nextchar is sorted */
387  break;
388  }
389  } else {
390  UNWIF_complist *i;
391  for (i = (UNWIF_complist *)cell; i->nextchar; i++) {
392  if (cp2 == i->nextchar)
393  return i->composite;
394  else if (cp2 < i->nextchar) /* nextchar is sorted */
395  break;
396  }
397  }
398  return 0;
399 }

406  return 0;
407  row = plane[(cp >> 8) & 0xff];
408  if (row)
409  return row[cp & 0xff];
410  else
411  return 0;

457 /* create an unordered decomposed wide string */
458 EXPORT errno_t
459 wcsnorm_decompose_s(wchar_t *restrict dest, rsize_t dmax, wchar_t *restrict src,
460  rsize_t *restrict lenp, bool iscompat)
461 {
462  rsize_t orig_dmax;
463  wchar_t *orig_dest;
464  const wchar_t *overlap_bumper;
465  wint_t cp;
466  int c;
467 
468  if (lenp)
469  *lenp = 0;
470 
471  if (unlikely(dest == NULL)) {
472  invoke_safe_str_constraint_handler("wcsnorm_s: "
473  "dest is null",
474  NULL, ESNULLP);
475  return RCNEGATE(ESNULLP);
476  }
477 
478  if (unlikely(src == NULL)) {
479  invoke_safe_str_constraint_handler("wcsnorm_s: "
480  "src is null",
481  NULL, ESNULLP);
482  *dest = 0;
483  return RCNEGATE(ESNULLP);
484  }
485 
486  if (unlikely(dmax == 0)) {
487  invoke_safe_str_constraint_handler("wcsnorm_s: "
488  "dmax is 0",
489  NULL, ESZEROL);
490  *dest = 0;
491  return RCNEGATE(ESZEROL);
492  }
493 
494  if (unlikely(dmax < 5)) {
495  invoke_safe_str_constraint_handler("wcsnorm_s: "
496  "dmax < 5",
497  NULL, ESLEMIN);
498  *dest = 0;
499  return RCNEGATE(ESLEMIN);
500  }
501 
502  if (unlikely(dmax > RSIZE_MAX_STR)) {
503  invoke_safe_str_constraint_handler("wcsnorm_s: "
504  "dmax exceeds max",
505  NULL, ESLEMAX);
506  *dest = 0;
507  return RCNEGATE(ESLEMAX);
508  }
509 
510  if (unlikely(dest == src)) {
511  invoke_safe_str_constraint_handler("wcsnorm_s: "
512  "overlapping objects",
513  NULL, ESOVRLP);
514  return RCNEGATE(ESOVRLP);
515  }
516 #ifndef HAVE_NORM_COMPAT
517  if (iscompat) {
518  invoke_safe_str_constraint_handler("wcsnorm_s: "
519  "not configured with --enable-norm-compat",
520  NULL, EOF);
521  *dest = 0;
522  return RCNEGATE(EOF);
523  }
524 #else
525  if (unlikely(iscompat && dmax < 19)) {
526  invoke_safe_str_constraint_handler("wcsnorm_s: "
527  "dmax is < 19",
528  NULL, ESLEMIN);
529  *dest = 0;
530  return RCNEGATE(ESLEMIN);
531  }
532 #endif
533 
534  /* hold base of dest in case src was not copied */
535  orig_dmax = dmax;
536  orig_dest = dest;
537 
538  if (dest < src) {
539  overlap_bumper = src;
540 
541  while (dmax > 0) {
542  cp = _dec_w16(src);
543  if (unlikely(dest == overlap_bumper)) {
544  handle_werror(orig_dest, orig_dmax, "wcsnorm_decompose_s: "
545  "overlapping objects",
546  ESOVRLP);
547  return RCNEGATE(ESOVRLP);
548  }
549  /* A surrogate pair can only represent max _UNICODE_MAX */
550 #if SIZEOF_WCHAR_T > 2
551  if (unlikely(_UNICODE_MAX < cp)) {
552  handle_werror(orig_dest, orig_dmax, "wcsnorm_decompose_s: "
553  "cp is too high",
554  ESLEMAX);
555  return ESLEMAX;
556  }
557 #endif
558  if (!cp)
559  goto done;
560 
561  c = _decomp_s(dest, dmax, cp, iscompat);
562  if (c > 0) {
563  dest += c;
564  dmax -= c;
565 #if SIZEOF_WCHAR_T == 2
566  if (cp > 0xffff)
567  src++;
568 #endif
569  } else if (c == 0) {
570 #if SIZEOF_WCHAR_T == 2
571  if (cp > 0xffff) {
572  *dest++ = *src++;
573  dmax--;
574  }
575 #endif
576  *dest++ = *src;
577  dmax--;
578  } else {
579  handle_werror(orig_dest, orig_dmax, "wcsnorm_decompose_s: "
580  "decomposition error",
581  -c);
582  return -c;
583  }
584  src++;
585  }
586  } else {
587  overlap_bumper = dest;
588 
589  while (dmax > 0) {
590  cp = _dec_w16(src);
591  if (unlikely(src == overlap_bumper)) {
592  handle_werror(orig_dest, orig_dmax, "wcsnorm_decompose_s: "
593  "overlapping objects",
594  ESOVRLP);
595  return RCNEGATE(ESOVRLP);
596  }
597 #if SIZEOF_WCHAR_T > 2
598  if (unlikely(_UNICODE_MAX < cp)) {
599  handle_werror(orig_dest, orig_dmax, "wcsnorm_decompose_s: "
600  "cp is too high",
601  ESLEMAX);
602  return ESLEMAX;
603  }
604 #endif
605  if (!cp)
606  goto done;
607 
608  c = _decomp_s(dest, dmax, cp, iscompat);
609  if (c > 0) {
610  dest += c;
611  dmax -= c;
612 #if SIZEOF_WCHAR_T == 2
613  if (cp > 0xffff)
614  src++;
615 #endif
616  } else if (c == 0) {
617 #if SIZEOF_WCHAR_T == 2
618  if (cp > 0xffff) {
619  *dest++ = *src++;
620  dmax--;
621  }
622 #endif
623  *dest++ = *src;
624  dmax--;
625  } else {
626  handle_werror(orig_dest, orig_dmax, "wcsnorm_decompose_s: "
627  "decomposition error",
628  -c);
629  return RCNEGATE(-c);
630  }
631  src++;
632  }
633  }
634 
635  if (lenp)
636  *lenp = orig_dmax - dmax;
637  handle_werror(orig_dest, orig_dmax, "wcsnorm_decompose_s: "
638  "dmax too small",
639  ESNOSPC);
640  return RCNEGATE(ESNOSPC);
641 
642  done:
643  if (lenp)
644  *lenp = orig_dmax - dmax;
645 #ifdef SAFECLIB_STR_NULL_SLACK
646  memset(dest, 0, dmax*sizeof(wchar_t));
647 #else
648  *dest = 0;
649 #endif
650  return EOK;
651 }

679 /* reorder decomposed sequence to NFD */
680 EXPORT errno_t
681 wcsnorm_reorder_s(wchar_t *restrict dest, rsize_t dmax, wchar_t *restrict p,
682  rsize_t len)
683 {
684  UNWIF_cc seq_ary[CC_SEQ_SIZE];
685  UNWIF_cc* seq_ptr = (UNWIF_cc*)seq_ary; /* start with stack */
686  UNWIF_cc* seq_ext = NULL; /* heap when needed */
687  size_t seq_max = CC_SEQ_SIZE;
688  size_t cc_pos = 0;
689  const wchar_t *e = p + len;
690  wchar_t *orig_dest = dest;
691  rsize_t orig_dmax = dmax;
692 
693  while (p < e) {
694  uint8_t cur_cc;
695  wint_t cp = _dec_w16(p);
696  p++;
697 #if SIZEOF_WCHAR_T == 2
698  if (cp > 0xffff)
699  p++;
700 #endif
701 
702  cur_cc = _combin_class(cp);
703  if (cur_cc != 0) {
704  if (seq_max < cc_pos + 1) { /* extend if need */
705  seq_max = cc_pos + CC_SEQ_STEP; /* new size */
706  if (CC_SEQ_SIZE == cc_pos) { /* seq_ary full */
707  seq_ext = (UNWIF_cc*)malloc(seq_max*sizeof(UNWIF_cc));
708  memcpy(seq_ext, seq_ary, cc_pos*sizeof(UNWIF_cc));
709  }
710  else {
711  seq_ext = (UNWIF_cc*)realloc(seq_ext, seq_max*sizeof(UNWIF_cc));
712  }
713  seq_ptr = seq_ext; /* use seq_ext from now */
714  }
715 
716  seq_ptr[cc_pos].cc = cur_cc;
717  seq_ptr[cc_pos].cp = cp;
718  seq_ptr[cc_pos].pos = cc_pos;
719  ++cc_pos;
720

794 /* combine decomposed sequences to NFC. */
795 /* iscontig = false; composeContiguous? FCC if true */
796 EXPORT errno_t
797 wcsnorm_compose_s(wchar_t *restrict dest, rsize_t dmax, wchar_t *restrict p,
798  rsize_t *restrict lenp, bool iscontig)
799 {
800  const wchar_t *e = p + *lenp;
801  wint_t cpS = 0; /* starter code point */
802  bool valid_cpS = false; /* if false, cpS isn't initialized yet */
803  uint8_t pre_cc = 0;
804 
805  wint_t seq_ary[CC_SEQ_SIZE];
806  wint_t* seq_ptr = (wint_t*) seq_ary; /* either stack or heap */
807  wint_t* seq_ext = NULL; /* heap */
808  size_t seq_max = CC_SEQ_SIZE;
809  size_t cc_pos = 0;
810  wchar_t *orig_dest = dest;
811  rsize_t orig_dmax = dmax;
812 
813  while (p < e) {
814  uint8_t cur_cc;
815  wint_t cp = _dec_w16(p);
816  p++;
817 #if SIZEOF_WCHAR_T == 2
818  if (cp > 0xffff)
819  p++;
820 #endif
821 
822  cur_cc = _combin_class(cp);
823 
824  if (!valid_cpS) {
825  if (cur_cc == 0) {
826  cpS = cp; /* found the first Starter */
827  valid_cpS = true;
828  if (p < e)
829  continue;
830  }
831  else {
832  _ENC_W16(dest, dmax, cp);
833  if (unlikely(!dmax)) {
834  handle_werror(orig_dest, orig_dmax, "wcsnorm_compose_s: "
835  "dmax too small",
836  ESNOSPC);
837  return RCNEGATE(ESNOSPC);
838  }
839  continue;
840  }
841  }
842  else {
843  bool composed;
844 
845  /* blocked */
846  if ((iscontig && cc_pos) || /* discontiguous combination (FCC) */
847  (cur_cc != 0 && pre_cc == cur_cc) || /* blocked by same CC */
848  (pre_cc > cur_cc)) /* blocked by higher CC: revised D2 */
849  composed = false;
850 
851  /* not blocked:
852  iscontig && cc_pos == 0 -- contiguous combination
853  cur_cc == 0 && pre_cc == 0 -- starter + starter
854  cur_cc != 0 && pre_cc < cur_cc -- lower CC */
855  else {
856  /* try composition */
857  wint_t cpComp = _composite_cp(cpS, cp);
858 
859  if (cpComp && !isExclusion(cpComp)) {
860  cpS = cpComp;
861  composed = true;
862 
863  /* pre_cc should not be changed to cur_cc */
864  /* e.g. 1E14 = 0045 0304 0300 where CC(0304) == CC(0300) */
865  if (p < e)
866  continue;
867  }
868  else
869  composed = false;
870  }
871 
872  if (!composed) {
873  pre_cc = cur_cc;
874  if (cur_cc != 0 || !(p < e)) {
875  if (seq_max < cc_pos + 1) { /* extend if need */
876  seq_max = cc_pos + CC_SEQ_STEP; /* new size */
877  if (CC_SEQ_SIZE == cc_pos) { /* seq_ary full */
878  seq_ext = (wint_t*)malloc(seq_max*sizeof(wint_t));
879  memcpy(seq_ext, seq_ary, cc_pos*sizeof(wint_t));
880  }
881  else {
882  seq_ext = (wint_t*)realloc(seq_ext, seq_max*sizeof(wint_t));
883  }
884  seq_ptr = seq_ext; /* use seq_ext from now */
885  }
886  seq_ptr[cc_pos] = cp;
887  ++cc_pos;
888  }
889  if (cur_cc != 0 && p < e)
890  continue;
891  }
892  }
893 
894  /* output */
895  _ENC_W16(dest, dmax, cpS); /* starter (composed or not) */
896  if (unlikely(!dmax)) {
897  handle_werror(orig_dest, orig_dmax, "wcsnorm_compose_s: "
898  "dmax too small",
899  ESNOSPC);
900  return RCNEGATE(ESNOSPC);
901  }
902 
903  if (cc_pos == 1) {
904  _ENC_W16(dest, dmax, *seq_ptr);
905  cc_pos = 0;
906  } else if (cc_pos > 1) {
907  memcpy(dest, seq_ptr, cc_pos*sizeof(wchar_t));
908  dest += cc_pos;
909  dmax -= cc_pos;
910  cc_pos = 0;
911  }
912 
913  cpS = cp;
914  }
915  if (seq_ext)
916  free(seq_ext);
917 
918 #ifdef SAFECLIB_STR_NULL_SLACK
919  memset(dest, 0, dmax*sizeof(wchar_t));
920 #else
921  *dest = 0;
922 #endif
923  *lenp = orig_dmax - dmax;
924  return EOK;
925 }
