342 EXPORT wint_t
343 _towcase(wint_t wc, int lower)
344 {
345  int i;
346  int lmul = 2*lower-1;
347  int lmask = lower-1;
348  /* no letters with case in these large ranges */
349  if (/*(!iswalpha(wc) is broken on some platforms) */
350  wc < 0x41
351  || (unsigned)wc - 0x0600 <= 0x0fff-0x0600
352  || (unsigned)wc - 0x2e00 <= 0xa63f-0x2e00
353  || (unsigned)wc - 0xa800 <= 0xab69-0xa800
354  || (unsigned)wc - 0xabc0 <= 0xfeff-0xabc0
355  )
356  return wc;
357  /* special case because the diff between upper/lower is too big */
358  if (lower && (unsigned)wc - 0x10a0 < 0x2e) {
359  if (wc>0x10c5 && wc != 0x10c7 && wc != 0x10cd) { return wc; }
360  else { return wc + 0x2d00 - 0x10a0; }
361  }
362  else if (!lower && (unsigned)wc - 0x2d00 < 0x26) {
363  if (wc>0x2d25 && wc != 0x2d27 && wc != 0x2d2d) return wc;
364  else return wc + 0x10a0 - 0x2d00;
365  }
366  for (i=0; casemaps[i].len; i++) {
367  int base = casemaps[i].upper + (lmask & casemaps[i].lower);
368  assert(i>0 ? casemaps[i].upper >= casemaps[i-1].upper : 1);
369  if ((unsigned)wc-base < casemaps[i].len) {
370  if (casemaps[i].lower == 1)
371  return wc + lower - ((wc-casemaps[i].upper)&1);
372  return wc + lmul*casemaps[i].lower;
373  }
374  if (lower && casemaps[i].upper > wc)
375  break;
376  }
377  for (i=0; pairs[i][1-lower]; i++) {
378  assert(i>0 ? pairs[i][0] >= pairs[i-1][0] : 1);
379  if (pairs[i][1-lower] == wc)
380  return pairs[i][lower];
381  if (lower && pairs[i][0] > wc)
382  break;
383  }
384  for (i=0; casemapsl[i].len; i++) {
385  unsigned long base = casemapsl[i].upper + (lmask & casemapsl[i].lower);
386  assert(i>0 ? casemapsl[i].upper >= casemapsl[i-1].upper : 1);
387  if ((unsigned)wc-base < casemapsl[i].len) {
388  if (casemapsl[i].lower == 1)
389  return wc + lower - ((wc-casemapsl[i].upper)&1);
390  return wc + lmul*casemapsl[i].lower;
391  }
392  if (lower && casemaps[i].upper > wc)
393  break;
394  }
395  return wc;
396 }
