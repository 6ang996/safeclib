49 timingsafe_memcmp (const void *b1, const void *b2, size_t len)
50 {
51  const unsigned char *p1 = (const unsigned char *)b1, *p2 = (const unsigned char *)b2;
52  size_t i;
53  int res = 0, done = 0;
54 
55  for (i = 0; i < len; i++) {
56  /* lt is -1 if p1[i] < p2[i]; else 0. */
57  int lt = (p1[i] - p2[i]) >> CHAR_BIT;
58 
59  /* gt is -1 if p1[i] > p2[i]; else 0. */
60  int gt = (p2[i] - p1[i]) >> CHAR_BIT;
61 
62  /* cmp is 1 if p1[i] > p2[i]; -1 if p1[i] < p2[i]; else 0. */
63  int cmp = lt - gt;
64 
65  /* set res = cmp if !done. */
66  res |= cmp & ~done;
67 
68  /* set done if p1[i] != p2[i]. */
69  done |= lt | gt;
70  }
71 
72  return (res);
73 }
