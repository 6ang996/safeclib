161 static void cycle(size_t width, unsigned char* ar[], int n)
162 {
163  unsigned char tmp[256];
164  size_t l;
165  int i;
166 
167  if (n < 2) {
168  return;
169  }
170 
171  ar[n] = tmp;
172  while (width) {
173  l = sizeof(tmp) < width ? sizeof(tmp) : width;
174  memcpy(ar[n], ar[0], l);
175  for (i = 0; i < n; i++) {
176  memcpy(ar[i], ar[i + 1], l);
177  ar[i] += l;
178  }
179  width -= l;
180  }

183 /* shl() and shr() need n > 0 */
184 static inline void shl(size_t p[2], int n)
185 {
186  if (n >= (int)(8 * sizeof(size_t))) {
187  n -= 8 * sizeof(size_t);
188  p[1] = p[0];
189  p[0] = 0;
190  }
191  p[1] <<= n;
192  p[1] |= p[0] >> (sizeof(size_t) * 8 - n);
193  p[0] <<= n;
194 }

208 static void sift(unsigned char *head, size_t width, cmpfun cmp, int pshift,
209  size_t lp[], void *ctx)
210 {
211  unsigned char *rt, *lf;
212  unsigned char *ar[14 * sizeof(size_t) + 1];
213  int i = 1;
214 
215  ar[0] = head;
216  while (pshift > 1) {
217  rt = head - width;
218  lf = head - width - lp[pshift - 2];
219 
220  if ( (*cmp)(ar[0], lf, ctx) >= 0 &&
221  (*cmp)(ar[0], rt, ctx) >= 0)
222  {
223  break;
224  }
225  if ((*cmp)(lf, rt, ctx) >= 0) {
226  ar[i++] = lf;
227  head = lf;
228  pshift -= 1;
229  } else {
230  ar[i++] = rt;
231  head = rt;
232  pshift -= 2;
233  }
234  }
235  cycle(width, ar, i);
236 }

238 static void trinkle(unsigned char *head, size_t width, cmpfun cmp, size_t pp[2],
239  int pshift, int trusty, size_t lp[], void *ctx)
240 {
241  unsigned char *stepson,
242  *rt, *lf;
243  size_t p[2];
244  unsigned char *ar[14 * sizeof(size_t) + 1];
245  int i = 1;
246  int trail;
247 
248  p[0] = pp[0];
249  p[1] = pp[1];
250 
251  ar[0] = head;
252  while (p[0] != 1 || p[1] != 0) {
253  stepson = head - lp[pshift];
254  if ((*cmp)(stepson, ar[0], ctx) <= 0) {
255  break;
256  }
257  if (!trusty && pshift > 1) {
258  rt = head - width;
259  lf = head - width - lp[pshift - 2];
260  if ( (*cmp)(rt, stepson, ctx) >= 0 ||
261  (*cmp)(lf, stepson, ctx) >= 0)
262  {
263  break;
264  }
265  }
266 
267  ar[i++] = stepson;
268  head = stepson;
269  trail = pntz(p);
270  shr(p, trail);
271  pshift += trail;
272  trusty = 0;
273  }
274  if (!trusty) {
275  cycle(width, ar, i);
276  sift(head, width, cmp, pshift, lp, ctx);
277  }

280 static void
281 qsort_musl(void *base, size_t nel, size_t width, cmpfun cmp, void *ctx)
282 {
283  size_t lp[12*sizeof(size_t)];
284  size_t i, size = width * nel;
285  unsigned char *head, *high;
286  size_t p[2] = {1, 0};
287  int pshift = 1;
288  int trail;
289 
290  if (!size) return;
291 
292  head = (unsigned char *)base;
293  high = head + size - width;
294 
295  /* Precompute Leonardo numbers, scaled by element width */
296  for (lp[0]=lp[1]=width, i=2; (lp[i]=lp[i-2]+lp[i-1]+width) < size; i++);
297 
298  while (head < high) {
299  if ((p[0] & 3) == 3) {
300  sift(head, width, cmp, pshift, lp, ctx);
301  shr(p, 2);
302  pshift += 2;
303  } else {
304  if (lp[pshift - 1] >= (size_t)(high - head)) {
305  trinkle(head, width, cmp, p, pshift, 0, lp, ctx);
306  } else {
307  sift(head, width, cmp, pshift, lp, ctx);
308  }
309 
310  if (pshift == 1) {
311  shl(p, 1);
312  pshift = 0;
313  } else {
314  shl(p, pshift - 1);
315  pshift = 1;
316  }
317  }
318 
319  p[0] |= 1;
320  head += width;
321  }
322 
323  trinkle(head, width, cmp, p, pshift, 0, lp, ctx);
324 
325  while (pshift != 1 || p[0] != 1 || p[1] != 0) {
326  if (pshift <= 1) {
327  trail = pntz(p);
328  shr(p, trail);
329  pshift += trail;
330  } else {
331  shl(p, 2);
332  pshift -= 2;
333  p[0] ^= 7;
334  shr(p, 1);
335  trinkle(head - lp[pshift] - width, width, cmp, p, pshift + 1, 1, lp, ctx);
336  shl(p, 1);
337  p[0] |= 1;
338  trinkle(head - width, width, cmp, p, pshift, 1, lp, ctx);
339  }
340  head -= width;
341  }
342 }
