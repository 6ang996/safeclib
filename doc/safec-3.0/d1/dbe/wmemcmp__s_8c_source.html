74 wmemcmp_s (const wchar_t *dest, rsize_t dmax,
75  const wchar_t *src, rsize_t smax, int *diff)
76 {
77  const wchar_t *dp;
78  const wchar_t *sp;
79 
80  dp = (wchar_t*) dest;
81  sp = (wchar_t*) src;
82 
83  /*
84  * must be able to return the diff
85  */
86  if (unlikely(diff == NULL)) {
87  invoke_safe_mem_constraint_handler("wmemcmp_s: diff is null",
88  NULL, ESNULLP);
89  return (RCNEGATE(ESNULLP));
90  }
91  *diff = -1; /* default diff */
92 
93  if (unlikely(dp == NULL)) {
94  invoke_safe_mem_constraint_handler("wmemcmp_s: dest is null",
95  NULL, ESNULLP);
96  return (RCNEGATE(ESNULLP));
97  }
98 
99  if (unlikely(sp == NULL)) {
100  invoke_safe_mem_constraint_handler("wmemcmp_s: src is null",
101  NULL, ESNULLP);
102  return (RCNEGATE(ESNULLP));
103  }
104 
105  if (unlikely(dmax == 0)) {
106  invoke_safe_mem_constraint_handler("wmemcmp_s: dmax is 0",
107  NULL, ESZEROL);
108  return (RCNEGATE(ESZEROL));
109  }
110 
111  if (unlikely(dmax > RSIZE_MAX_WMEM || smax > RSIZE_MAX_WMEM)) {
112  invoke_safe_mem_constraint_handler("wmemcmp_s: dmax/smax exceeds max",
113  NULL, ESLEMAX);
114  return (RCNEGATE(ESLEMAX));
115  }
116 
117  if (unlikely(smax == 0)) {
118  invoke_safe_mem_constraint_handler("wmemcmp_s: smax is 0",
119  NULL, ESZEROL);
120  return (RCNEGATE(ESZEROL));
121  }
122 
123  if (unlikely(smax > dmax)) {
124  invoke_safe_mem_constraint_handler("wmemcmp_s: smax exceeds dmax",
125  NULL, ESNOSPC);
126  return (RCNEGATE(ESNOSPC));
127  }
128 
129  /*
130  * no need to compare the same memory
131  */
132  if (unlikely(dp == sp)) {
133  *diff = 0;
134  return (RCNEGATE(EOK));
135  }
136 
137  /*
138  * now compare sp to dp
139  */
140  *diff = 0;
141  while (dmax > 0 && smax > 0) {
142  if (*dp != *sp) {
143  /*** *diff = *dp - *sp; ***/
144  *diff = *dp < *sp ? -1 : 1;
145  break;
146  }
147 
148  dmax--;
149  smax--;
150 
151  dp++;
152  sp++;
153  }
154 
155  return (RCNEGATE(EOK));
156 }
157 EXPORT_SYMBOL(wmemcmp_s)
