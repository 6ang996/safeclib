186 /* Return the number of wide lower-case characters needed to full fold-case
187  the given uppercase character. Returns 0, 1, 2 or 3.
188  0 if the charcater stays the same, 1 if one character changes,
189  2 or 3 if the character will be replaced with 2 or 3.
190 
191  Note that accents expand to more characters than 1 via NFD decomposition.
192  iswfc only matches towfc_s, but not wcsfc_s which does NFD decomposition also.
193 */
194 int
195 iswfc(wint_t wc)
196 {
197  /* the slow variant would walk the 2 loops */
198  if (likely((wc < 0xdf) ||
199  (wc > 0x0587 && wc < 0x1e96) ||
200  (wc > 0x1FFC && wc < 0xFB00) ||
201  (wc > 0xFB17) ))
202  goto single;
203  if (wc < 0x1e96) {
204  if (wc == 0xdf ||
205  wc == 0x130 ||
206  wc == 0x149 ||
207  wc == 0x1f0 ||
208  wc == 0x587)
209  return 2;
210  else if (wc == 0x390 ||
211  wc == 0x3b0)
212  return 3;
213  else
214  goto single;
215  }
216  if (wc <= 0x1e9a || wc == 0x1e9e || wc == 0x1f50)
217  return 2;
218  if (wc < 0x1f50)
219  goto single;
220  if (wc == 0x1f52 || wc == 0x1f54 || wc == 0x1f56)
221  return 3;
222  if (wc < 0x1f80)
223  goto single;
224  /* Note these ranges 1f80 - 1faf deviate in fc from lc */
225  if (wc <= 0x1faf || (wc >= 0x1fb2 && wc < 0x1fb6)) {
226  if (wc == 0x1fb5) goto single;
227  else return 2;
228  }
229  /* XXX 1f80- expand via nfd => 3-4
230  if (wc <= 0x1faf) {
231  if ((wc & 0x7) == 0 || (wc & 0x7) == 1)
232  return 2;
233  else
234  return 3;
235  }
236  if (wc >= 0x1fb2 && wc < 0x1fb6)
237  return 2;
238  */
239  if (wc == 0x1fb7 || wc == 0x1fc7 || wc == 0x1fd2 || wc == 0x1fd3 || wc == 0x1fd7 ||
240  wc == 0x1fe2 || wc == 0x1fe3 || wc == 0x1fe7 || wc == 0x1ff7)
241  return 3;
242  if (wc == 0x1fb5) goto single;
243  else if (0)
244  return 2;
245  if (wc == 0x1fc5) goto single;
246  if (wc == 0x1fb6 || wc == 0x1fbc ||
247  (wc >= 0x1fc2 && wc <= 0x1fc6) ||
248  wc == 0x1fcc ||
249  wc == 0x1fd6 ||
250  wc == 0x1fe4 ||
251  wc == 0x1fe6 ||
252  (wc >= 0x1ff2 && wc <= 0x1ff4) ||
253  wc == 0x1ff6 ||
254  wc == 0x1ffc)
255  return 2;
256  if (wc < 0xfb00 || wc > 0xfb17 || (wc > 0xfb06 && wc < 0xfb13))
257  goto single;
258  if (wc == 0xfb03 || wc == 0xfb04)
259  return 3;
260  return 2;
261 
262  if (0) {
263  single:
264  return iswupper(wc) ? 1 : 0;
265  }
266 }

268 /* The 194 single fc chars where fc is different to lc.
269  Must only be called when we know for sure that the length is 1!
270  Returns 1 if fc is different to lc, else -ESNOTFND.
271 
272  perl5.27.3 -E'no warnings; for (0..0x10ffff){
273  my ($lc,$fc) = (lc(pack"W",$_), fc(pack"W",$_));
274  printf "U+%04X: fc: %X, lc: %X\n", $_, unpack("W",$fc), unpack("W",$lc)
275  if $lc ne $fc and length($fc)==1;
276  }'
277  */
278 int
279 _towfc_single(wchar_t *restrict dest, const wint_t src)
280 {
281  /* fc exceptions: not towlower */
282  dest[1] = L'\0';
283  if (src < 0xb5) goto single;
284  if (src >= 0xb5 && src <= 0x3f5) {
285  switch(src) {
286  case 0xb5: dest[0] = 0x3bc; return EOK; /* lc b5 */
287  case 0x17f: dest[0] = 0x73; return EOK; /* lc 17f */
288  case 0x345: dest[0] = 0x3b9; return EOK; /* lc 345 */
289  case 0x3c2: dest[0] = 0x3c3; return EOK; /* lc 3c2 */
290  case 0x3d0: dest[0] = 0x3b2; return EOK; /* lc 3d0 */
291  case 0x3d1: dest[0] = 0x3b8; return EOK; /* lc 3d1 */
292  case 0x3d5: dest[0] = 0x3c6; return EOK; /* lc 3d5 */
293  case 0x3d6: dest[0] = 0x3c0; return EOK; /* lc 3d6 */
294  case 0x3f0: dest[0] = 0x3ba; return EOK; /* lc 3f0 */
295  case 0x3f1: dest[0] = 0x3c1; return EOK; /* lc 3f1 */
296  case 0x3f5: dest[0] = 0x3b5; return EOK; /* lc 3f5 */
297  }
298  goto single;
299  }
300  else if (unlikely( src >= 0x13a0 && src <= 0x13f5 )) {
301  dest[0] = src; return EOK;
302  }
303  else if (unlikely( src >= 0x13f8 && src <= 0x13fd )) {
304  dest[0] = src - 8; return EOK;
305  }
306  else if (unlikely( src <= 0x1c88 )) {
307  if (src < 0x1c80) goto single;
308  switch (src) {
309  case 0x1c80: dest[0] = 0x432; return EOK;
310  case 0x1c81: dest[0] = 0x434; return EOK;
311  case 0x1c82: dest[0] = 0x43e; return EOK;
312  case 0x1c83: dest[0] = 0x441; return EOK;
313  case 0x1c84: dest[0] = 0x442; return EOK;
314  case 0x1c85: dest[0] = 0x442; return EOK;
315  case 0x1c86: dest[0] = 0x44a; return EOK;
316  case 0x1c87: dest[0] = 0x463; return EOK;
317  case 0x1c88: dest[0] = 0xa64b; return EOK;
318  }
319  }
320  else if (unlikely( src <= 0x1fbe )) {
321  if (src < 0x1e9b) goto single;
322  if (src == 0x1e9b) { dest[0] = 0x1e61; return EOK; }
323  if (src == 0x1fbe) { dest[0] = 0x3b9; return EOK; }
324  goto single;
325  }
326  else if (unlikely( src >= 0xab70 && src <= 0xabbf )) {
327  dest[0] = src - (0xab70-0x13a0); return EOK;
328  }
329  /*
330  else if (unlikely( (src >= 0xff21 && src <= 0xff3a) ||
331  (src >= 0x10400 && src <= 0x104d3) ||
332  (src >= 0x10c80 && src <= 0x10cb2) ||
333  (src >= 0x118a0 && src <= 0x118bf) ||
334  (src >= 0x118a0 && src <= 0x118bf) )) {
335  dest[0] = src + 0x20;
336  return EOK;
337  }
338  else if (unlikely( src >= 0x1e900 && src <= 0x1e921 )) {
339  dest[0] = src + 0x22;
340  return EOK;
341  }
342  */
343 
344  single:
345  dest[0] = src < 128 ? tolower(src) : _towcase(src, 1);
346  return (wint_t)dest[0] == src ? -(ESNOTFND) : 1;
347 }

380 /* Writes the fold-cased wide string to dest from the given uppercase wide
381  character. dmax should be 4 (3 wchar's + \0).
382  Returns the number of replaced wide characters, or -ESNOTFND if not replaced.
383 */
384 int
385 towfc_s(wchar_t *restrict dest, rsize_t dmax, const wint_t src)
386 {
387  int i;
388 
389  if (unlikely(dest == NULL)) {
390  invoke_safe_str_constraint_handler("_towfc_s: "
391  "dest is null",
392  NULL, ESNULLP);
393  return -(ESNULLP);
394  }
395  if (unlikely(dmax < 4)) {
396  invoke_safe_str_constraint_handler("_towfc_s: "
397  "dmax is < 4",
398  NULL, ESLEMIN);
399  return -(ESLEMIN);
400  }
401  dest[0] = L'\0';
402  if (unlikely(dmax > RSIZE_MAX_STR)) {
403  invoke_safe_str_constraint_handler("_towfc_s: "
404  "dmax exceeds max",
405  NULL, ESLEMAX);
406  return -(ESLEMAX);
407  }
408 
409  if (src < 128) {
410  dest[1] = L'\0';
411  dest[0] = tolower(src);
412  return (wint_t)dest[0] == src ? -(ESNOTFND) : 1;
413  }
414 
415  for (i=0; tbl2[i].upper; i++) {
