101 EXPORT errno_t
102 wcsfc_s(wchar_t *restrict dest, rsize_t dmax, wchar_t *restrict src,
103  rsize_t *restrict lenp)
104 {
105  wchar_t *orig_dest;
106  rsize_t orig_dmax;
107  int is_lithuanian = 0;
108  int is_tr_az = 0;
109 
110  if (lenp)
111  *lenp = 0;
112  if (unlikely(dest == NULL)) {
113  invoke_safe_str_constraint_handler("wcsfc_s: "
114  "dest is null",
115  NULL, ESNULLP);
116  return (ESNULLP);
117  }
118 
119  if (unlikely(dmax == 0)) {
120  invoke_safe_str_constraint_handler("wcsfc_s: "
121  "dmax is 0",
122  NULL, ESZEROL);
123  return (ESZEROL);
124  }
125 
126  if (unlikely(dmax > RSIZE_MAX_STR)) {
127  invoke_safe_str_constraint_handler("wcsfc_s: "
128  "dmax exceeds max",
129  NULL, ESLEMAX);
130  return (ESLEMAX);
131  }
132 
133  if (unlikely(src == NULL)) {
134  handle_werror(dest, dmax, "wcsfc_s: " "src is null",
135  ESNULLP);
136  return (ESNULLP);
137  }
138 
139  if (unlikely(
140  (*src >= 0x49 && *src <= 0x130) &&
141  (*src == 0x49 ||
142  *src == 0x4A ||
143  *src == 0x130 ||
144  *src == 0xcc ||
145  *src == 0xcd ||
146  *src == 0x128 ||
147  *src == 0x12e))) {
148  const char *lc = setlocale(LC_CTYPE, NULL);
149  if (lc && *lc) {
150  if ( (*lc == 't' && *(lc+1) == 'r') ||
151  (*lc == 'a' && *(lc+1) == 'z') )
152  is_tr_az = 1;
153  else
154  if (*lc == 'l' && *(lc+1) == 't')
155  is_lithuanian = 1;
156  }
157  }
158 
159  orig_dest = dest;
160  orig_dmax = dmax;
161 
162  while (*src && dmax > 0) {
163  wchar_t tmp[4];
164  wint_t cp = _dec_w16((wchar_t *)src);
165  int c = iswfc(cp);
166 #if SIZEOF_WCHAR_T == 2
167  if (cp > 0xffff)
168  src++;
169 #endif
170  if (unlikely(c > 1)) {
171  /* can this be further decomposed? */
172  errno_t rc = towfc_s(tmp, 4, cp);
173  if (rc < 0)
174  return rc;
175  /* I-Dot for Turkish and Azeri */
176  if (unlikely(is_tr_az && *src == 0x130)) {
177  *dest++ = 0x69; dmax--; /* skip the \x307 dot */
178  } else if (unlikely(*src >= 0x1f80 && *src <= 0x1ff4)) {
179  /* only some greek PROSGEGRAMMENI + YPOGEGRAMMENI need to be decomposed further */
180  /* decompose mult. chars */
181  wchar_t tmpd[8];
182  int i;
183  for (i=0; i<c; i++) {
184  int d;
185  wint_t cp1 = _dec_w16(&tmp[i]);
186 #if SIZEOF_WCHAR_T > 2
187  if (unlikely(_UNICODE_MAX < cp1)) {
188  handle_werror(orig_dest, orig_dmax, "wcsfc_s: "
189  "cp is too high",
190  ESLEMAX);
191  return ESLEMAX;
192  }
193 #endif
194  d = _decomp_s(tmpd, 8, cp1, false);
195  if (d) { /* decomp. max 4 */
196  memcpy(dest, tmpd, d*sizeof(wchar_t));
197  dest += d;
198  dmax -= d;
199  } else {
200  _ENC_W16(dest, dmax, cp1);
201  }
202  }
203  } else {
204  memcpy(dest, tmp, c*sizeof(wchar_t));
205  dest += c;
206  dmax -= c;
207  }
208  src++;
209  } else { /* c = 0 or 1. 1 might still be special case */
210  if (unlikely(*src == 0x3a3)) {
211  if (iswspace(*(src+1))) /* final sigma? */
212  *dest++ = 0x3c2;
213  else
214  *dest++ = 0x3c3;
215  src++;
216  dmax--;
217  }
218  else if (unlikely(is_lithuanian)) {
219  /* I-Dot/J-Dot for Lithuanian, I-Dot for Turkish and Azeri.
220  http://unicode.org/reports/tr21/tr21-5.html#SpecialCasing */
221  switch (*src) {
222  case 0xcc:
223  *dest++ = 0x69;
224  *dest++ = 0x307;
225  *dest++ = 0x300;
226  dmax -= 3;
227  break;
228  case 0xcd:
229  *dest++ = 0x69;
230  *dest++ = 0x307;
231  *dest++ = 0x301;
232  dmax -= 3;
233  break;
234  case 0x128:
235  *dest++ = 0x69;
236  *dest++ = 0x307;
237  *dest++ = 0x303;
238  dmax -= 3;
239  break;
240  case 0x49: /* Add the dot with accented I's or J's */
241  case 0x4A:
242  /* lt More_Above */
243  if (_is_lt_accented(*(src+1))) {
244  /*towfc_s(tmp, 4, 0x0130);*/
245  *dest++ = (*src + 0x20);
246  *dest++ = 0x307;
247  dmax -= 2;
248  src++;
249  break;
250  }
251  /* FALLTHRU */
252  case 0x12e:
253  if (_is_lt_accented(*(src+1))) {
254  *dest++ = 0x12f;
255  *dest++ = 0x307;
256  dmax -= 2;
257  src++;
258  break;
259  }
260  /* FALLTHRU */
261  default:
262  goto is_single;
263  }
264  src++;
265  }
266  else if (unlikely(is_tr_az)) {
267  /* pair I-Dot with i for Turkish and Azeri.
268  http://unicode.org/reports/tr21/tr21-5.html#SpecialCasing */
269  if (*src == 0x130) {
270  *dest++ = 0x307; src++; dmax--;
271  }
272  /* Remove dot_above in the sequence
273  I + dot_above, which will turn into i */
274  else if (*src == L'I' && *(src+1) == 0x307) {
275  *dest++ = L'i'; dmax--;
276  src++; src++;
277  }
278  /* Unless an I is before a dot_above, it turns
279  into a dotless i. */
280  else if (*src == 0x49 &&
281  (*(src+1) != 0x307 ||
282  *(src+1) != 0x130)) {
283  *dest++ = 0x131; src++; dmax--;
284  } else {
285  is_single:
286  (void)_towfc_single(dest, _dec_w16((wchar_t*)src));
287  src++;
288  /* even if not found dest[0] still contains towlower */
289  dest++;
290  dmax--;
291  }
292  } else {
293  if (unlikely(dmax < 5))
294  goto too_small;
295  (void)_towfc_single(tmp, _dec_w16((wchar_t*)src));
296  src++;
297  if (tmp[0] >= 0xc0) {
298  cp = _dec_w16(tmp);
299  c = _decomp_s(dest, dmax, cp, false);
300  }
301  else
302  c = 0;
303  if (!c) {
304  *dest++ = tmp[0];
305  dmax--;
306  } else {
307  /* eg accents 101 => 61 304 */
308  dest += c;
309  dmax -= c;
310  }
311  }
312  }
313  }
314  /* write the length even if too small,
315  so the client knows how mush is needed */
316  if (lenp)
317  *lenp = orig_dmax - dmax;
318  if (unlikely(dmax <= 0)) {
319  too_small:
320  handle_werror(orig_dest, orig_dmax,
321  "wcsfc_s: " "dmax too small",
322  ESNOSPC);
323  return (ESNOSPC);
324  }
325 
326 #ifdef SAFECLIB_STR_NULL_SLACK
327  memset(dest, 0, dmax*sizeof(wchar_t));
328  /*while (dmax) { *dest = '\0'; dmax--; dest++; }*/
329 #else
330  *dest = 0;
331 #endif
332 
333  return (EOK);
334 }
